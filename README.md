Data Cleaning & Preparation

```sql
-- Remove duplicates using ROW_NUMBER()
WITH duplicate_cte AS (
  SELECT *, ROW_NUMBER() OVER(
    PARTITION BY company, location, industry, total_laid_off, 
    percentage_laid_off, date, stage, country, funds_raised_millions
  ) AS Row_num
  FROM layoffs
)

```

Complete SQL Analysis Code

```sql
-- Data Cleaning Process
SELECT * FROM layoffs;

-- Creating a backup 
CREATE TABLE layoffs_row LIKE layoffs;
INSERT INTO layoffs_row SELECT * FROM layoffs;

-- Remove duplicates
CREATE TABLE layoffs AS
SELECT *, ROW_NUMBER() OVER(
  PARTITION BY company, location, industry, total_laid_off, 
  percentage_laid_off, date, stage, country, funds_raised_millions
) AS row_num FROM layoffs1;

DELETE FROM layoffs WHERE row_num > 1;

-- Standardize data
UPDATE layoffs SET company = TRIM(company);
UPDATE layoffs SET industry = 'Crypto' WHERE industry LIKE 'Crypto%';
UPDATE layoffs SET country = 'United States' WHERE country LIKE 'United States%';
UPDATE layoffs SET date = STR_TO_DATE(date, '%m/%d/%Y');

-- Handle null values
UPDATE layoffs t1 
JOIN layoffs t2 ON t1.company = t2.company 
SET t1.industry = t2.industry
WHERE t1.industry IS NULL AND t2.industry IS NOT NULL;

DELETE FROM layoffs 
WHERE total_laid_off IS NULL AND percentage_laid_off IS NULL;

ALTER TABLE layoffs DROP COLUMN row_num;

-- Exploratory Data Analysis EDA
SELECT MAX(total_laid_off), MAX(percentage_laid_off) FROM layoffs;

-- Top companies
SELECT company, SUM(total_laid_off) 
FROM layoffs GROUP BY company ORDER BY 2 DESC;

-- Top industries
SELECT industry, SUM(total_laid_off) 
FROM layoffs GROUP BY industry ORDER BY 2 DESC;

-- Top countries
SELECT country, SUM(total_laid_off) 
FROM layoffs GROUP BY country ORDER BY 2 DESC;

-- Yearly trends
SELECT YEAR(date), SUM(total_laid_off) 
FROM layoffs GROUP BY YEAR(date);

-- Monthly rolling totals
WITH Rolling_total AS (
  SELECT SUBSTRING(date, 1, 7) AS MONTH, SUM(total_laid_off) AS total_off
  FROM layoffs
  WHERE SUBSTRING(date, 1, 7) IS NOT NULL
  GROUP BY MONTH
)
SELECT MONTH, total_off,
SUM(total_off) OVER(ORDER BY MONTH) AS Rolling_total
FROM Rolling_total;

-- Top 5 companies each year
WITH company_year AS (
  SELECT company, YEAR(date) as years, SUM(total_laid_off) as total_laid_off
  FROM layoffs GROUP BY company, YEAR(date)
),
Company_Year_Rank AS (
  SELECT *, DENSE_RANK() OVER(PARTITION BY years ORDER BY total_laid_off DESC) AS Ranking
  FROM company_year WHERE years IS NOT NULL
)
SELECT * FROM Company_Year_Rank WHERE Ranking <= 5 ORDER BY 2,4;

-- Top 5 industries each year
WITH Industry_year AS (
  SELECT industry, YEAR(date) as years, SUM(total_laid_off) as total_laid_off
  FROM layoffs GROUP BY industry, YEAR(date)
),
Industry_Year_Rank AS (
  SELECT *, DENSE_RANK() OVER(PARTITION BY years ORDER BY total_laid_off DESC) AS Ranking
  FROM Industry_year WHERE years IS NOT NULL
)
SELECT * FROM Industry_Year_Rank WHERE Ranking <= 5 ORDER BY 2,4;
```
